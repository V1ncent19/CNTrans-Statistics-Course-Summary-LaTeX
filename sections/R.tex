\begin{lstlisting}[language=R]
ggplot(data = , aes(x = , y = , ...)) +
    geom_XXX(...) + stat_XXX(...) + ... +
    annotate(...)+ ... +
    scale_XXX(...) + coord_XXX(...) + guides(...) + theme(...)    
\end{lstlisting}
